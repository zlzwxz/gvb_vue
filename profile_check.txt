<template>
  <div class="profile-view">
    <div class="container">
      <div class="profile-grid">
        <!-- дёӘдәәдҝЎжҒҜеҚЎзүҮ -->
        <div class="profile-card card">
          <div class="avatar-area">
            <div class="avatar-circle">
              {{ (userStore.userInfo?.nick_name || userStore.userInfo?.user_name || 'U')[0].toUpperCase() }}
            </div>
            <h2 class="user-name">{{ userStore.userInfo?.nick_name || userStore.userInfo?.user_name }}</h2>
            <p class="user-bio">{{ userStore.userInfo?.bio || 'иҝҷдёӘдәәеҫҲжҮ’пјҢд»Җд№ҲйғҪжІЎз•ҷдё? }}</p>
          </div>

          <!-- дҝ®ж”№жҳөз§°иЎЁеҚ• -->
          <el-divider>дҝ®ж”№дёӘдәәдҝЎжҒҜ</el-divider>
          <el-form :model="profileForm" label-width="80px" class="profile-form">
            <el-form-item label="жҳөз§°">
              <el-input v-model="profileForm.nick_name" placeholder="иҜ·иҫ“е…Ҙжҳөз§? />
            </el-form-item>
            <el-form-item label="дёӘдәәзӯҫеҗҚ">
              <el-input v-model="profileForm.bio" type="textarea" :rows="3" placeholder="дёҖеҸҘиҜқд»Ӣз»ҚиҮӘе·ұ" />
            </el-form-item>
            <el-form-item label="дёӘдәәй“ҫжҺҘ">
              <el-input v-model="profileForm.link" placeholder="https://..." />
            </el-form-item>
            <el-form-item>
              <el-button type="primary" :loading="profileLoading" @click="saveProfile">дҝқеӯҳдҝ®ж”№</el-button>
            </el-form-item>
          </el-form>

          <!-- з»‘е®ҡйӮ®з®ұ -->
          <el-divider>з»‘е®ҡйӮ®з®ұ</el-divider>
          <el-form label-width="80px">
            <el-form-item label="йӮ®з®ұ">
              <el-input v-model="emailForm.email" placeholder="иҜ·иҫ“е…ҘйӮ®з®ұең°еқҖ" />
            </el-form-item>
            <el-form-item>
              <el-button type="primary" plain :loading="emailLoading" @click="bindEmail">з»‘е®ҡйӮ®з®ұ</el-button>
            </el-form-item>
          </el-form>
        </div>
      </div>
    </div>
  </div>
</template>

<script setup>
import { reactive, ref, onMounted } from 'vue'
import { ElMessage } from 'element-plus'
import { useUserStore } from '@/stores/user'
import { apiUpdateUserNickName, apiBindEmail } from '@/api/user'

const userStore = useUserStore()

// дҝ®ж”№жҳөз§°иЎЁеҚ•
const profileForm = reactive({
  nick_name: '',
  bio: '',
  link: ''
})

// з»‘е®ҡйӮ®з®ұиЎЁеҚ•
const emailForm = reactive({ email: '' })

const profileLoading = ref(false)
const emailLoading = ref(false)

// еҲқе§ӢеҢ–иЎЁеҚ•еҶ…е®?
onMounted(() => {
  const info = userStore.userInfo
  if (info) {
    profileForm.nick_name = info.nick_name || ''
    profileForm.bio = info.bio || ''
    profileForm.link = info.link || ''
  }
})

/**
 * дҝқеӯҳдёӘдәәдҝЎжҒҜпјҲжҳөз§°гҖҒзӯҫеҗҚгҖҒй“ҫжҺҘпјү
 */
const saveProfile = async () => {
  profileLoading.value = true
  try {
    await apiUpdateUserNickName(profileForm)
    await userStore.fetchUserInfo()
    ElMessage.success('дҝ®ж”№жҲҗеҠҹпј?)
  } finally {
    profileLoading.value = false
  }
}

/**
 * з»‘е®ҡйӮ®з®ұ
 */
const bindEmail = async () => {
  if (!emailForm.email) {
    ElMessage.warning('иҜ·иҫ“е…ҘйӮ®з®ұең°еқҖ')
    return
  }
  emailLoading.value = true
  try {
    await apiBindEmail(emailForm)
    ElMessage.success('йӮ®з®ұз»‘е®ҡжҲҗеҠҹпј?)
  } finally {
    emailLoading.value = false
  }
}
</script>
